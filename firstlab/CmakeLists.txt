cmake_minimum_required(VERSION 3.10)
project(CppLabExperiments)

# Устанавливаем стандарт C++
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Создаем три отдельных исполняемых файла
add_executable(sort_demo sort.cpp sort.h)
add_executable(int_demo int.cpp int_utils.h)
add_executable(float_demo float.cpp float_utils.h)

# Если хочешь, чтобы выводились предупреждения (полезно для обучения)
if(MSVC)
    target_compile_options(sort_demo PRIVATE /W4)
    target_compile_options(int_demo PRIVATE /W4)
    target_compile_options(float_demo PRIVATE /W4)
else()
    target_compile_options(sort_demo PRIVATE -Wall -Wextra)
    target_compile_options(int_demo PRIVATE -Wall -Wextra)
    target_compile_options(float_demo PRIVATE -Wall -Wextra)
endif()
